# Nginx configuration for sm.aeseo1.org
# SSE and Proxy support enabled

server {
    listen 80;
    listen [::]:80;
    server_name sm.aeseo1.org;

    # Redirect HTTP to HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    listen [::]:443 ssl http2;
    server_name sm.aeseo1.org;

    # SSL Configuration (Let's Encrypt)
    ssl_certificate /etc/letsencrypt/live/sm.aeseo1.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/sm.aeseo1.org/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;

    # Security headers
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;
    add_header X-XSS-Protection "1; mode=block" always;
    add_header Referrer-Policy "no-referrer-when-downgrade" always;

    # Max body size for large requests
    client_max_body_size 10M;

    # Access logs
    access_log /var/log/nginx/sm.aeseo1.org-access.log;
    error_log /var/log/nginx/sm.aeseo1.org-error.log;

    # Root directory for frontend
    root /home/peter/sitemap-crawler-webapp/frontend/dist;
    index index.html;

    # ============================================
    # Backend API (Flask on port 8000)
    # ============================================
    location /api/ {
        # Proxy to Flask backend
        proxy_pass http://127.0.0.1:8000;

        # Standard proxy headers
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # Important timeouts for long-running requests
        proxy_connect_timeout 300s;
        proxy_send_timeout 300s;
        proxy_read_timeout 300s;

        # Disable buffering for real-time responses
        proxy_buffering off;

        # HTTP/1.1 for keep-alive
        proxy_http_version 1.1;
        proxy_set_header Connection "";
    }

    # ============================================
    # SSE Streaming Endpoint (CRITICAL for real-time)
    # ============================================
    location /api/crawl-stream {
        # Proxy to Flask backend
        proxy_pass http://127.0.0.1:8000;

        # Standard headers
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        # *** SSE CRITICAL SETTINGS ***
        # Disable buffering (MUST HAVE for SSE)
        proxy_buffering off;

        # Disable caching
        proxy_cache off;

        # Keep connection alive
        proxy_http_version 1.1;
        proxy_set_header Connection "";

        # Extended timeouts for long-running streams
        proxy_connect_timeout 3600s;  # 1 hour
        proxy_send_timeout 3600s;
        proxy_read_timeout 3600s;

        # SSE specific headers
        proxy_set_header X-Accel-Buffering no;  # Disable nginx buffering

        # Content type for SSE
        add_header Cache-Control "no-cache, no-store, must-revalidate";
        add_header Pragma "no-cache";
        add_header Expires "0";
        add_header X-Accel-Buffering "no";
    }

    # ============================================
    # Frontend (React SPA)
    # ============================================
    location / {
        try_files $uri $uri/ /index.html;

        # Cache static assets
        location ~* \.(js|css|png|jpg|jpeg|gif|ico|svg|woff|woff2|ttf|eot)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
        }

        # No cache for HTML
        location ~* \.html$ {
            expires -1;
            add_header Cache-Control "no-cache, no-store, must-revalidate";
        }
    }

    # Health check endpoint (no auth required)
    location /api/health {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        access_log off;  # Don't log health checks
    }

    # Block access to sensitive files
    location ~ /\. {
        deny all;
        access_log off;
        log_not_found off;
    }

    location ~ ~$ {
        deny all;
        access_log off;
        log_not_found off;
    }
}
